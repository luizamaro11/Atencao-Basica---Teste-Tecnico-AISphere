flowchart TD

%% ENTRY
A[Usuário acessa o sistema] --> B{Está autenticado?}

%% AUTH
B -- Não --> C[Tela de Login]
C --> D[Usuário insere email e senha]
D --> E{Credenciais válidas?}

E -- Não --> F[Exibir erro: email ou senha inválidos]
F --> C

E -- Sim --> G[Redirecionar para Dashboard]

%% REGISTER
C --> H[Ir para cadastro]
H --> I[Formulário de registro]
I --> J[Preencher nome, email, senha]
J --> K{Dados válidos?}

K -- Não --> L[Exibir erros de validação]
L --> I

K -- Sim --> M[Criar usuário]
M --> N[Redirecionar para login]

%% DASHBOARD
B -- Sim --> G
G --> O[Dashboard]

O --> P[Menu principal]

%% NAVIGATION
P --> Q[Ver pacientes]
P --> R[Logout]

R --> S[Encerrar sessão]
S --> C

%% LISTAGEM
Q --> T[Listagem de pacientes]
T --> U{Tem pacientes?}

U -- Não --> V[Exibir estado vazio]
V --> W[Botão: Cadastrar paciente]

U -- Sim --> X[Exibir tabela de pacientes]

%% SEARCH
X --> Y[Campo de busca]
Y --> Z[Usuário digita termo]
Z --> AA[Filtrar lista em tempo real]

%% ACTIONS
X --> AB[Ações por paciente]
AB --> AC[Editar]
AB --> AD[Excluir]

%% CREATE
W --> AE[Formulário de cadastro]
AE --> AF[Preencher dados do paciente]
AF --> AG{Dados válidos?}

AG -- Não --> AH[Exibir erros]
AH --> AE

AG -- Sim --> AI[Salvar paciente]
AI --> AJ[Exibir sucesso]
AJ --> T

%% EDIT
AC --> AK[Formulário de edição]
AK --> AL[Alterar dados]
AL --> AM{Dados válidos?}

AM -- Não --> AN[Exibir erros]
AN --> AK

AM -- Sim --> AO[Atualizar paciente]
AO --> AP[Exibir sucesso]
AP --> T

%% DELETE
AD --> AQ[Confirmar exclusão?]

AQ -- Não --> T
AQ -- Sim --> AR[Remover paciente (soft delete)]
AR --> AS[Exibir sucesso]
AS --> T

%% LOADING STATES
T --> AT[Loading ao carregar dados]
AE --> AT
AK --> AT

%% ERROR STATES
AI --> AU{Erro no servidor?}
AU -- Sim --> AV[Exibir erro genérico]
AV --> AE

AO --> AU
AR --> AU
